# JDBC 使用说明

# JDBC 编程步骤

# 1. 加载驱动程序：

Class.forName(driverClass)
//加载MySql驱动
Class.forName("com.mysql.jdbc.Driver")
//加载Oracle驱动
Class.forName("oracle.jdbc.driver.OracleDriver")

# 2. 获得数据库连接：

DriverManager.getConnection("jdbc:mysql://localhost:3306/fruit?useUnicode=true&characterEncoding=utf8&useSSL=false", "root", "000000");

# 3. 创建Statement或者PreparedStatement对象：

conn.createStatement();
conn.prepareStatement(sql);

# 完整实例

package com.seecen;

import java.sql.*;

public class Dao {

	private static final String URL = "jdbc:mysql://localhost:3306/fruit?useUnicode=true&characterEncoding=utf8&useSSL=false";
	private static final String USER = "root";
	private static final String PASS = "000000";

	// 1.加载驱动程序
	static {
		try {
			Class.forName("com.mysql.jdbc.Driver");
		} catch (ClassNotFoundException e) {
			e.printStackTrace();
		}
	}

	// 2. 获得数据库连接
	protected Connection get() {
		try {
			return DriverManager.getConnection(URL, USER, PASS);
		} catch (SQLException e) {
			e.printStackTrace();
		}
		return null;
	}

	// 3. 构造函数
	public Dao() throws SQLException {
        Statement stmt = get().createStatement();
        ResultSet rs = stmt.executeQuery("SELECT * FROM t_fruit");
        //如果有数据，rs.next()返回true
        while(rs.next()){
            System.out.println("编号："+rs.getString("number")+"，名称："+rs.getString("name"));
        }
	}

	public static void main(String[] args) throws SQLException {
		new Dao();

	}

}

# 增删改查

package com.seecen;

import java.sql.*;

public class CRUD extends Dao {

	public CRUD() throws SQLException {
		Fruit fruit = new Fruit();
		fruit.setNumber(99999);
		fruit.setName("橘子");
		int result = editFruit(fruit);

		if(result > 0) {
			System.out.println("add success");
		}
	}

	public static void main(String[] args) throws SQLException {
		new CRUD();
		
	}

	// 增加
	public int addFruit(Fruit fruit) throws SQLException {

		int row = 0;
		
		// 获取连接
		Connection conn = get();

		String sql = "INSERT INTO t_fruit(number, name, price, unit) values(?,?,?,?)";

		// 预编译
		PreparedStatement ptmt = conn.prepareStatement(sql); // 预编译SQL，减少sql执行

		// 传参
		ptmt.setInt(1, fruit.getNumber());
		ptmt.setString(2, fruit.getName());
		ptmt.setDouble(3, fruit.getPrice());
		ptmt.setString(4, fruit.getUnit());

		// 执行
		row = ptmt.executeUpdate();
		
		
		//返回结果
		return row;
	}
	
	// 修改
	public int editFruit(Fruit fruit) throws SQLException {
		
		int row = 0;
		
		// 获取连接
		Connection conn = get();

		String sql = "UPDATE t_fruit SET name = ? WHERE number = ? ";

		// 预编译
		PreparedStatement ptmt = conn.prepareStatement(sql); 

		// 传参
		ptmt.setString(1, fruit.getName());
		ptmt.setInt(2, fruit.getNumber());

		// 执行
		row = ptmt.executeUpdate();

		//返回结果
		return row;
		
	}
	
	// 删除
	public int delFruit(Fruit fruit) throws SQLException {
		int row = 0;
		
		// 获取连接
		Connection conn = get();

		String sql = "DELETE FROM t_fruit WHERE number = ? ";

		// 预编译
		PreparedStatement ptmt = conn.prepareStatement(sql); 

		// 传参
		ptmt.setInt(1, fruit.getNumber());

		// 执行
		row = ptmt.executeUpdate();

		//返回结果
		return row;
	}

}

# 封装数据库方法

public class Dao {

	private static final String URL = "jdbc:mysql://localhost:3306/fruit?useUnicode=true&characterEncoding=utf8&useSSL=false";
	private static final String USER = "root";
	private static final String PASS = "000000";

	static {
		try {
			Class.forName("com.mysql.jdbc.Driver");
		} catch (ClassNotFoundException e) {
			e.printStackTrace();
		}
	}
	
	protected Connection get() {
        try {
            return DriverManager.getConnection(URL, USER, PASS);
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return null;
	}
	
    public int saveOrUpdateOrDelete(String sql, Object... params) {
        Connection con = null;
        PreparedStatement pstm = null;
        int row = 0;
        try {
            con = get();
            pstm = con.prepareStatement(sql);
            for (int i = 0; i < params.length; i++) {
                pstm.setObject(i + 1, params[i]);
            }
            row = pstm.executeUpdate();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            release(null, pstm, con);
        }
        return row;
    }
	
    protected void release(ResultSet rs, Statement stmt, Connection con) {
        try {
            if (rs != null && !rs.isClosed()) {
                rs.close();
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        try {
            if (stmt != null && !stmt.isClosed()) {
                stmt.close();
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        try {
            if (con != null && !con.isClosed()) {
                con.close();
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
    
}

# 用LIST保存数据

public class FruitDao extends Dao {

	public List<Fruit> queryFruit() {

		List<Fruit> list = null;
		String sql = "select * from t_fruit";
		ResultSet rs = null;
		PreparedStatement pstm = null;
		Connection conn = null;
		try {
			conn = get();
			pstm = conn.prepareStatement(sql);
			rs = pstm.executeQuery();
			while (rs.next()) {
				if (rs.isFirst()) {
					list = new ArrayList<>();
				}
				Fruit fruit = new Fruit();
				fruit.setNumber(rs.getInt("number"));
				fruit.setName(rs.getString("name"));
				fruit.setPrice(rs.getFloat("price"));
				fruit.setUnit(rs.getString("unit"));
				list.add(fruit);
			}
		} catch (Exception e) {
			e.printStackTrace();
		} finally {
			release(rs, pstm, conn);
		}
		return list;
	}

}
